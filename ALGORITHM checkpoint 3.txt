PROCEDURE dot_product(v1 ,v2)
VAR
    ps: INTEGER:=0
    i: INTEGER:=0
BEGIN
    FOR i FROM 0 TO V1.length STEP 1  DO
        ps=ps+V1[i]*V2[i]
    END_FOR
END

FUNCTION dot_product(v1 ,v2) : INTEGER
VAR
    ps: INTEGER:=0
    i: INTEGER:=0
BEGIN
    FOR i FROM 0 TO V1.length STEP 1  DO
        ps=ps+V1[i]*V2[i]
    END_FOR
    RETURN ps ;
